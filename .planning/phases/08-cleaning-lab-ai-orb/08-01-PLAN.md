       ```swift

       }
       ```

    3. Update `labelColor` computed property:
       ```swift
       private var labelColor: Color {
           switch (variant, tint) {
           case (.filled, .accent): return .white
           case (.filled, .ai):     return Color(uiColor: SonicMergeTheme.ColorPalette.primaryText) // #1C1C1E — 7.38:1 AAA on Lime Green
           case (.outline, _):      return Color(uiColor: semantic.textPrimary)
           }
       }
       ```
       CRITICAL: White on Lime Green = 1.98:1 WCAG fail. Dark #1C1C1E on Lime Green = 7.38:1 AAA pass.

    4. Update `backgroundFill` computed property:
       ```swift
       @ViewBuilder
       private var backgroundFill: some View {
           switch (variant, tint) {
           case (.filled, .accent):
               Color(uiColor: semantic.accentAction).opacity(isEnabled ? 1.0 : 0.35)
           case (.filled, .ai):
               Color(uiColor: semantic.accentAI).opacity(isEnabled ? 1.0 : 0.35)
           case (.outline, _):
               Color.clear
           }
       }
       ```

    5. Update `borderOverlay` computed property to branch on tint for outline:

       ```swift
       @ViewBuilder
       private var borderOverlay: some View {

                   : Color(uiColor: semantic.accentAction)
               Capsule()
                   .strokeBorder(
                       strokeColor.opacity(isEnabled ? 0.5 : 0.2),
                       lineWidth: 1
                   )
                   .allowsHitTesting(false)
           }
       }
       ```
