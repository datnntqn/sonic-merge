    ```swift
    import Foundation
    import SwiftData

    @Model
    final class AudioClip {
        var id: UUID
        var displayName: String
        // Stored as relative path component (filename only) — reconstruct absolute URL at runtime
        // via AppConstants.clipsDirectory().appending(path: fileURLRelativePath)
        var fileURLRelativePath: String
        var duration: TimeInterval
        var sampleRate: Double
        var channelCount: Int
        var importedAt: Date
        var sortOrder: Int

        init(displayName: String, fileURLRelativePath: String, duration: TimeInterval) {
            self.id = UUID()
            self.displayName = displayName
            self.fileURLRelativePath = fileURLRelativePath
            self.duration = duration
            self.sampleRate = 48_000
            self.channelCount = 2
            self.importedAt = .now
            self.sortOrder = 0
        }

        // Computed property — not stored in SwiftData (computed from App Group container at runtime)
        var fileURL: URL {
            get throws {
                try AppConstants.clipsDirectory().appending(path: fileURLRelativePath)
            }
        }
    }
    ```

    NOTE on fileURL storage: Per RESEARCH.md Pattern 2 critical note, absolute App Group paths can shift between device and simulator. Store only the filename component (e.g., "A1B2C3-....m4a") as `fileURLRelativePath`. Reconstruct the full URL at runtime via `AppConstants.clipsDirectory()`. This makes PersistenceTests work correctly with in-memory stores because the model does not hold device-specific paths.

    Create SonicMerge/App/AppConstants.swift:

    ```swift
    import Foundation

    enum AppConstants {
        static let appGroupID = "group.com.yourteam.SonicMerge"

        static func clipsDirectory() throws -> URL {
            guard let container = FileManager.default.containerURL(
                forSecurityApplicationGroupIdentifier: appGroupID
            ) else {
                throw AppGroupError.containerNotFound
            }
            let dir = container.appending(path: "clips", directoryHint: .isDirectory)
            try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
            return dir
        }
    }

    enum AppGroupError: Error {
        case containerNotFound
    }
    ```

    Rewrite SonicMerge/SonicMergeApp.swift (replacing the Xcode template):

    ```swift
    import SwiftUI
    import SwiftData
    import AVFAudio

    @main
    struct SonicMergeApp: App {

        let modelContainer: ModelContainer = {
            let schema = Schema([AudioClip.self])
            let config = ModelConfiguration(
                schema: schema,
                groupContainer: .identifier(AppConstants.appGroupID)
            )
            do {
                return try ModelContainer(for: schema, configurations: [config])
            } catch {
                // Fatal: if the SwiftData store cannot be created the app cannot function.
                // This fails only if the App Group entitlement is missing from the target.
                // Fix: Xcode > target > Signing & Capabilities > App Groups > add group.com.yourteam.SonicMerge
                fatalError("Failed to create ModelContainer: \(error). Check App Group entitlement.")
            }
        }()

        init() {
            configureAudioSession()
        }

        var body: some Scene {
            WindowGroup {
                // ImportView is provided by Plan 04. Placeholder until then.
                ContentView()
            }
            .modelContainer(modelContainer)
        }

        private func configureAudioSession() {
            do {
                // .playback: app imports pre-recorded files; never records live audio.
                // .mixWithOthers: allows background music to continue during import.
                // Activating at launch avoids audio "pop" when session starts mid-import.
                try AVAudioSession.sharedInstance().setCategory(
                    .playback,
                    mode: .default,
                    options: [.mixWithOthers]
                )
                try AVAudioSession.sharedInstance().setActive(true)
            } catch {
                // Non-fatal: normalization (AVAssetWriter to disk) does not require an active session.
                // Playback (Phase 2) will retry session activation before first playback.
                print("[SonicMergeApp] AVAudioSession configuration failed: \(error)")
            }
        }
    }
    ```

    IMPORTANT: The App Group entitlement must be added manually in Xcode before this compiles cleanly on a real device. In the simulator, `containerURL(forSecurityApplicationGroupIdentifier:)` returns nil without the entitlement — the fatalError will trigger. Add the checkpoint below to flag this.
